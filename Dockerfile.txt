# ใช้ Python base image แบบ slim
FROM python:3.11-slim

# ตั้ง working directory
WORKDIR /app

# ติดตั้งระบบ dependencies สำหรับ OpenCV และ EasyOCR
RUN apt-get update && apt-get install -y \
    build-essential \
    libgl1 \
    libglib2.0-0 \
    && rm -rf /var/lib/apt/lists/*

# คัดลอกไฟล์ requirements.txt ไปที่ container
COPY requirements.txt .

# ติดตั้ง Python packages แบบไม่เก็บ cache (เพื่อลดขนาด)
RUN pip install --no-cache-dir -r requirements.txt

# คัดลอกไฟล์แอปทั้งหมดไปยัง container
COPY . .

# สั่งรัน Flask app
CMD ["python", "app.py"]
